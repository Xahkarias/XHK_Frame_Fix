slot_giga_frameworld_food = {
    included_zone_sets = {
        food
    }

    exclude = {
        xhk_acot_zone_DM_food
        xhk_acot_zone_DE_food
        xhk_acot_zone_PE_food
        xhk_acot_zone_SE_food
    }

    unlock = { # planet scope
        custom_tooltip = {
            fail_text = zone_unavailable_tech_food_processing_1
            OR = {
                AND = {
                    exists = owner
                    owner = { is_anglers_empire = yes }
                    is_wet = yes
                }
                AND = {
                    exists = owner
                    owner = {
                        has_valid_civic = civic_agrarian_idyll
                    }
                }
                AND = {
                    exists = owner
                    owner = {
                        has_tech_or_functional_civic = {
                            TECH = tech_food_processing_1
                        }
                    }
                }
            }
        }
    }
}
